    >>> from seishub.util.tree import Tree
    >>> root=Tree()

# Tree.add
    >>> root.add('/node', [])
    >>> root.add('/blub/blah', 1)
    >>> root.add('/blub/muh', 'hello')

# overwriting previous node
    >>> root.add('/blub/muh', 'hallo')

# Tree.__str__
    >>> str(root)
    "{'node': [], 'blub': {'blah': 1, 'muh': 'hallo'}}"

# Tree.get
    >>> root.get('/')
    {'node': [], 'blub': {'blah': 1, 'muh': 'hallo'}}
    >>> root.get('//')
    {'node': [], 'blub': {'blah': 1, 'muh': 'hallo'}}
    >>> root.get('')
    {'node': [], 'blub': {'blah': 1, 'muh': 'hallo'}}
    >>> root.get()
    {'node': [], 'blub': {'blah': 1, 'muh': 'hallo'}}
    >>> root.get('/blub')
    {'blah': 1, 'muh': 'hallo'}
    >>> root.get('/blub/blah')
    1
    >>> root.get('/blub/blah/')
    1
    >>> root.get('blub/muh/')
    'hallo'
    >>> root.get(['blub','muh'])
    'hallo'
    >>> root.get('blub/muh/maeh/schuwps')
    'hallo'

# Tree.isLeaf
    >>> root.get('blub3/muh/')
   
    >>> root.isLeaf('/blub/muh')
    True
    >>> root.isLeaf('/muh')
    False
    >>> root.isLeaf()
    False
